const apiKey = '7191ffb5-91d7-4fb7-bc30-94f1b854ee6e'; const cardId = 'xy1-1'; // Example Card ID for Venusaur-EX const baseUrl = 'https://api.pokemontcg.io/v2/cards'; /** * Fetches card data by a specific Card ID. * @param {string} id - The unique ID of the card (e.g., 'xy1-1'). */ async function lookupCard(id) { const url = `${baseUrl}/${id}`; try { const response = await fetch(url, { headers: { 'X-Api-Key': apiKey, 'Content-Type': 'application/json' } }); // Handle HTTP errors if (!response.ok) { throw new Error(`HTTP error! Status: ${response.status}`); } const data = await response.json(); // The card data is nested under the 'data' field const card = data.data; if (card) { console.log(`✅ Card Found: ${card.name}`); console.log(` Set: ${card.set.name}`); console.log(` Rarity: ${card.rarity}`); console.log(` HP: ${card.hp}`); // You can use card.images.large for the image URL console.log(` Image URL: ${card.images.small}`); return card; } else { console.log(`❌ Card with ID ${id} not found.`); return null; } } catch (error) { console.error('An error occurred during lookup:', error.message); } } // --- Run the lookup --- lookupCard(cardId);
